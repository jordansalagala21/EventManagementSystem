upstream backend {
    server backend:8000;  # 8000 is the port where backend service is running
}
server {
  # port where nginx is running
  listen 80;

  # Max upload limit is unlimited
  client_max_body_size 200M;

    #  gzip compression
    gzip on;
    gzip_disable "msie6";

    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_buffers 16 8k;
    gzip_http_version 1.1;
    gzip_min_length 0;
    gzip_types text/plain application/javascript text/css application/json application/x-javascript text/xml application/xml application/xml+rss text/javascript application/vnd.ms-fontobject application/x-font-ttf font/opentype;
    # End of gzip compression


    location / {
      proxy_pass http://backend;
      proxy_set_header Host $http_host;
#     proxy_set_header X-Forwarded-Proto https;   # Uncomment if ssl is used
      proxy_set_header X-Url-Scheme $scheme;
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
      proxy_redirect off;

      # Increase the timeout values
      proxy_connect_timeout 300s;
      proxy_read_timeout 300s;
    }

    location /static/ {
      # Since static files are handled by nginx, cross origin needs to be handled by nginx
      add_header Access-Control-Allow-Origin *;
      alias /static/;
    }

    location /media/ {
      # Since static files are handled by nginx, cross origin needs to be handled by nginx
      add_header Access-Control-Allow-Origin *;
      alias /media/;
    }

}
